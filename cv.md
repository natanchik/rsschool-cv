def rot13(message):
    lst1 = [i for i in range(65, 91)]
    lst2 = [i for i in range(97, 123)]    
    ans = ""
    for s in message:
        if s.isalpha():
            lst = (lst1 if ord(s) in lst1 else lst2)
            s = (chr(ord(s)+13) if ord(s)+13 in lst else chr(ord(s)-13))
        ans += s    
    return ans
